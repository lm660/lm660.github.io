排序算法

  • 排序算法(Sorting algorithm)是一种能将一串资料依照特定排序方式排列的算法。

常见排序算法

  • 冒泡排序
  • 选择排序
  • 插入排序
  • 归并排序
  • 快速排序
  • 堆排序

排序算法复杂度

名称 时间复杂度 空间复杂度 描述
冒泡 O(n²) O(1) 两两比较相邻元素并交换位置，一趟排序后最大值移到数组最后面
选择 O(n²) O(1) 先设第一个元素为最小值，再从未排序元素中找到最小值交换。再设第二个元素为最小值……
插入 O(n²) O(1) 把无序的第一个元素插入到前面有序区的合适位置
归并 O(nlogn) O(n) 把数组分为两段，从两段中逐个选最小的元素移入到新数据段的末尾
快速 O(nlogn)|O(n²) O(logn) 选择一个基准元素，将数组分成左右两部分，左边的元素都小于或等于基准元素，右边的元素都大于基准元素。然后，对左右两部分分别进行递归调用快速排序，最终将整个数组排序。
堆 O(nlogn) O(1) 使用最大堆来维护一个有序序列，然后不断选择出最大的值。

1.冒泡排序

//最好O(N)->最坏O(N²)->平均O(N²)
 function swap(arr:number[],i:number,j:number){
  const temp=arr[i]
  arr[i]=arr[j]
  arr[j]=temp
}

function bubbleSort(arr:number[]):number[]{
  let n=arr.length
  
  //外层循环：控制多少轮，n个数需要n-1轮。0~n-2
  for(let i=0;i<n-1;i++){
    let isSwap=false//用来优化（假如本来就是正确的顺序，不用进行交换还要进行遍历，浪费性能）
    //内层循环：是找到这一轮中的最大值。
    for(let j=0;j<n-1-i;j++){//为什么是n-1？假如为n的话，表示j可以取到n-1，j+1就能取到n，但是数组是从开始，索引只到length-1。即j+1只能到n-1，所以j只能到n-2，所以是<n-1.
      if(arr[j]>arr[j+1]){
       // [arr[j],arr[j+1]]=[arr[j+1],arr[j]] //也可以交换
         swap(arr,j,j+1)
         isSwap=true
      }
    }

    if(!isSwap) break//表明某一轮下来没有进行一次交换，说明此时顺序已经正确了，不用再进行下面的循环了，停止扫描（）表示
  }
  return arr
}

2.选择排序

//最好O(N²)->最坏O(N²)->平均O(N²)
 function swap(arr:number[],i:number,j:number){
  const temp=arr[i]
  arr[i]=arr[j]
  arr[j]=temp
}
//先假设第一个值为最小值，然后将剩下的所有元素和设定的比较，找到比设定还小的，然后将这两个元素位置互换。这样一轮结束。然后第二轮假设就是第二个值为最小值了。。。
function selectionSort(arr:number[]):number[]{
  let n=arr.length
  for(let i=0;i<n-1;i++){//外层循环：经过多少轮的 找最小值（从0到n-2，有n-1轮，假如就剩最后两个元素
    let minIndex=i//保留最小元素的索引
    for(let j=1+i;j<n;j++){//内层循环：每次找到最小值。因为要找到最后一个元素，所以j要取到n-1即小于n
      if(arr[j]<arr[minIndex]){//当遇到一个比设定的还小的，就更新最小索引
        minIndex=j
      }
    }//一轮结束找到了最小的索引，然后将设定的和最小索引的元素交换
    if(i!==minIndex){//只要不相等才交换，如果假设的正好是最小值（minIndex没有被重新赋值）就不用交换了=》优化
    swap(arr,i,minIndex)//交换数组中两个位置的元素
    }
  }
  return arr
}

function insertionSOrt(arr :number[]):number[]{
  let n=arr.length
  for(let i=1;i<n;i++){//外层循环：n-1轮，因为默认第一个是有序的
    //内层循环
    const newNum=arr[i]//因为第一个默认是有序的，所以从第二个数开始为1
    let j=i-1//拿到前面一个值的索引
    while(arr[j]>newNum&&j>=0){//因为不知道插入到前面有序的第几个后面，所以用while循环，终止条件是前面的某一个值要比newNum小
      arr[j+1]=arr[j]//假如不比newNum小，让当前这个值往后移一位
      j--
    }
    arr[j+1]=newNum//找到索引为j的值比newNum小，所以在j后面插入newNum，因为循环中对j进行了--，所以这里要加1
  }
  return arr

}

4.归并排序

//最好O(logN)->最坏O(NlogN)->平均O(NlogN)

 function swap(arr:number[],i:number,j:number){
  const temp=arr[i]
  arr[i]=arr[j]
  arr[j]=temp
}

function mergeSort(arr:number[]):number[]{
  if(arr.length<=1)return arr//此时arr是一个个只包含一个元素的数组

  //1.分解（divide）：对数组仅分解（分解成两个小数组）
  //1.1切割数组
  const mid=Math.floor(arr.length/2)
  const leftArr=arr.slice(0,mid)//包含0但不包含mid
  const rightArr=arr.slice(mid)//从mid开始剩下的值

  //1.2递归切割leftArr和rightArr
  const newLeftArr=mergeSort(leftArr)
  const newRightArr=mergeSort(rightArr)

  //2.合并（merge）：将两个子数组进行合并（双指针）
  let newArr:number[]=[]
  //2.1定义双指针
  let i=0
  let j=0
  //循环比较左右两边数组哪个值小一些先加入到新数组里
  while(i<newLeftArr.length&&j<newRightArr.length){//指针不能超过子数组长度
    if(newLeftArr[i]<=newRightArr[j]){//判断小的一个添加到新的数组里面，然后让对应的指针加1
      newArr.push(newLeftArr[i])
      i++
    }else{
      newArr.push(newRightArr[j])
      j++
    }
  }
  //2.2判断某一个数组中是否还有剩余元素
  //循环完左边还有元素
  if(i<newLeftArr.length){
    newArr.push(...newLeftArr.slice(i))//将剩余的元素都添加到新数组里
  }
  //循环完右边还有元素
  if(j<newRightArr.length){
    newArr.push(...newRightArr.slice(j))
  }

  //3.将合并后的数组返回
  return newArr
}
let arr=[59,66,2,51,30,6,88]
const newnums=mergeSort(arr)
console.log(newnums);//[ 2, 6, 30, 51, 59, 66, 88]
console.log(arr);//[59,66,2,51,30,6,88]

5.快速排序

//冒泡排序的一种改进
//最好O(logN)->最坏O(N²)->平均O(NlogN)

 function swap(arr:number[],i:number,j:number){
  const temp=arr[i]
  arr[i]=arr[j]
  arr[j]=temp
}

function quickSort(arr:number[]):number[]{
  partition(0,arr.length-1)
  function partition(left:number,right:number){
    if(left>=right)return//当传进来是同一个索引，表明只要一个元素是就结束了
    //1.定义一个基准元素，最后一个元素
    const pivot=arr[right]
    //2.双指针进行交换操作（左边都是比pivot小的数，右边都是比pivot大的数）
    let i=left
    let j=right-1
    while(i<=j){//直到i>j了说明两个指针交叉了
      //当选用最后一个元素为基准元素时一定先从左边找到比pivot大的数
      while(arr[i]<pivot){
        i++
      }
      //找到停止，再从右边找到一个比pivot小的数
      while(arr[j]>pivot){
        j--
      }
      //此时已经找到比pivot大的i的元素以及比pivot小的j的元素
      if(i<=j){
        swap(arr,i,j)
        i++//左指针向右移动一位
        j--//右指针向左移动一位
      }
    }
    //将pivot放到正确的位置
    swap(arr,i,right)
    //至此一次划分完成

    //左右继续划分区域
    partition(left,i-1)
    partition(i+1,right)
  }
  return arr
}

let arr=[59,66,2,51,30,6,88]
const newnums=quickSort(arr)
console.log(newnums);//[ 2, 6, 30, 51, 59, 66, 88]
console.log(arr);//[ 2, 6, 30, 51, 59, 66, 88]

//O(NlogN)
 function swap(arr:number[],i:number,j:number){

  arr[i]=arr[j]
  arr[j]=temp

function heapSort(arr:number[]):number[]{
//1.获取数组长度
const n=arr.length
//2.对arr数组进行原地建堆
//2.1从第一个非叶子节点进行下滤操作
const start=Math.floor(n/2-1)
for(let i=start;i>=0;i--){
  //2.2进行下滤操作
  heapfidownDown(arr,n,i)
}
//3.对最大堆进行排序操作
for(let i=n-1;i>0;i--){
  swap(arr,0,i)
  heapfidownDown(arr,i,0)
}

 console.log(arr);

  return arr
}
//进行下滤操作的时候需要进行操作的数组，以及数组的长度，以及哪个位置进行下滤操作
function heapfidownDown(arr:number[],n:number,index:number){
  while(2*index+1<n){//左子节点存在
    let leftChildIndex=2*index+1
    let rightChildIndx=2*index+2
    let largreIndex=leftChildIndex
    if(rightChildIndx<n&&arr[rightChildIndx]>=arr[largreIndex]){
      largreIndex=rightChildIndx
    }
    if(arr[index]>=arr[largreIndex]){
      break
    }
    swap(arr,largreIndex,index)
    index=largreIndex
  }
}
