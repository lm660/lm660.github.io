实现call/apply/bind

call apply bind
JS系列
发布日期:   2023-07-19

1.作用

改变函数运行时的this指向。

举一个需要改变this的情况

var name = "lucy";
var obj = {
    name: "martin",
    say: function () {
        console.log(this.name);
    }
};
obj.say(); // martin，this 指向 obj 对象
setTimeout(obj.say,0); // lucy，this 指向 window 对象

当把say放在setTimeout方法中，在定时器中是作为回调函数来执行的，因此回到主栈执行时是在全局执行上下文的环境中执行的，这时候this指向window，所以输出lucy。实际需要的是this指向obj对象，这时候就需要该改变this指向了。

setTimeout(obj.say.bind(obj),0); //martin，this指向obj对象

2.区别

apply

apply接受两个参数，第一个参数是this的指向，第二个参数是函数接受的参数，以数组的形式传入。改变this指向后原函数会立即执行，且此方法只是临时改变this指向一次。当第一个参数为null、undefined的时候，默认指向window(在浏览器中)

function fn(...args){
    console.log(this,args);
}
let obj = {
    myname:"张三"
}
fn.apply(obj,[1,2]); // this会变成传入的obj，传入的参数必须是一个数组

fn(1,2) // 不绑定this指向window

fn.apply(null,[1,2]); // this指向window
fn.apply(undefined,[1,2]); // this指向window

call

call方法的第一个参数也是this的指向，后面传入的是一个参数列表。跟apply一样，改变this指向后原函数会立即执行，且此方法只是临时改变this指向一次。同样的，当第一个参数为null、undefined的时候，默认指向window(在浏览器中)

function fn(...args){
    console.log(this,args);
}
let obj = {
    myname:"张三"
}

fn.call(obj,1,2); // this会变成传入的obj，传入的参数必须是一个数组；
fn(1,2) // this指向window

fn.call(null,[1,2]); // this指向window
fn.call(undefined,[1,2]); // this指向window

bind

bind方法和call很相似，第一参数也是this的指向，后面传入的也是一个参数列表(但是这个参数列表可以分多次传入)。改变this指向后不会立即执行，而是返回一个永久改变this指向的函数

function fn(...args){
    console.log(this,args);
}
let obj = {
    myname:"张三"
}

const bindFn = fn.bind(obj); // this 也会变成传入的obj ，bind不是立即执行需要执行一次
bindFn(1,2) // this指向obj
fn(1,2) // this指向window

小结：

  • 三者都可以改变函数的this对象指向
  • 三者第一个参数都是this要指向的对象，如果如果没有这个参数或参数为undefined或null，则默认指向全局window
  • 三者都可以传参，但是apply是数组，而call是参数列表，且apply和call是一次性传入参数，而bind可以分为多次传入
  • bind是返回绑定this之后的函数，apply、call 则是立即执行

3.实现

call

Function.prototype.mycall = function(thisArg, ...args) {
  // 1.获取需要被执行的函数
  var fn = this
  // 2.对thisArg转成对象类型(防止它传入的是非对象类型)
  thisArg = (thisArg !== null && thisArg !== undefined) ? Object(thisArg): window
  // 3.调用需要被执行的函数
  thisArg.fn = fn
  var result = thisArg.fn(...args)
  delete thisArg.fn
  // 4.将最终的结果返回出去
  return result
}

function foo() {
  console.log(this)
}
function sum(num1, num2) {
  console.log(this, num1, num2)
  return num1 + num2
}

foo.mycall(undefined)//window
var result = sum.mycall("abc", 20, 30)//"abc",20,30
console.log(result)//50

apply

Function.prototype.myapply=function(thisArg,arrayArg){//这里不用剩余参数因为apply要求传的就是一个数组
    thisArg=(thisArg!==null||thisArg!==undefined)?Object(thisArg):window
    var fn=this
    thisArg.fn=fn
    //异或操作，arrArg有值就赋值给它，没有值就为空数组，不处理的话会报错，因为不传就是undefined，对undefined进行扩展不正确
    //为什么call的时候使用剩余参数就不会有问题呢，因为无论传不传剩余参数都至少有一个数组。
    arrayArg=arrayArg||[]
    var result=thisArg.fn(...arrayArg)
    return result
}
function num(num1,num2){
    console.log(this,num1,num2);
    return num1+num2
}
//apply调用的时候后面的参数要是一个数组
var result=num.myapply({name:'why'},[10,20])//{name:'why'} 20 30
console.log(result);//50

bind

//由bind的使用可知，bind是要返回一个函数的，这样才能再次调用，而且可以在调用bind的时候传递一些参数，在接收后调用再传一部分参数
Function.prototype.mybind=function(thisArg,...argArray){
    var fn=this
    thisArg=(thisArg!==null||thisArg!==undefined)?Object(thisArg):window
    function proxy(...args){
        thisArg.fn=fn
        var finallArg=[...argArray,...args]
        var result=thisArg.fn(...finallArg)
        return result
    }
    return proxy
}
function foo(num1,num2,num3,num4){
    console.log(num1,num2,num3,num4);
}
var bar=foo.mybind('aaa',10,20)
var result=bar(30,40)//'aaa' 10 20 30 40
console.log(result)//100

小结：

  1. 处理绑定的thisBing：不是对象的转换成对象；
  2. 获取要执行的函数；
  3. 执行函数：判断有没有传参；
  4. 返回结果

注意：this指的是要调用的函数。thisArg表示的是给这个函数绑定一个this，通过隐式绑定实现的。thisArg.fn()代表的是对这个函数的调用。
