继承

无标签
JS系列
发布日期:   2023-07-20

1.是什么

继承是面向对象的三大特征之一。

  • 优点：继承可以使得子类具有父类别的各种属性和方法，减少重复代码的数量

2.实现方式

原型链继承

// 父类: 公共属性和方法
function Person() {
  this.name = "why"
  this.friends = []
}
Person.prototype.eating = function() {
  console.log(this.name + " eating~")
}
// 子类: 特有属性和方法
function Student() {
  this.sno = 111
}
var p = new Person()
Student.prototype = p
Student.prototype.studying = function() {
  console.log(this.name + " studying~")
}

var stu = new Student()
console.log(stu.name)//why
stu.eating()//why eating~
stu.studying()//why studying~

// 原型链实现继承的弊端:
// 1.第一个弊端: 打印stu对象, 继承的属性是看不到的
 console.log(stu)//{sno: 111}

// 2.第二个弊端: 创建出来两个stu的对象
var stu1 = new Student()
var stu2 = new Student()
// 直接修改对象上的属性, 是给本对象添加了一个新属性
stu1.name = "kobe"
console.log(stu2.name)//why
// 获取引用, 修改引用属性中的值, 会相互影响
stu1.friends.push("kobe")
console.log(stu1.friends)//[ 'kobe' ]
console.log(stu2.friends)//[ 'kobe' ]

构造函数继承

借助 call调用Parent函数

function Person(){
    this.name = 'Person';
}

Person.prototype.getName = function () {
    return this.name;
}

function Student(){
    Person.call(this);
    this.type = 'student'
}

let student = new Student();
console.log(student);  // 没问题 { name: 'Person', type: 'student' }
console.log(student.getName());  // 会报错

//父类的引用属性不会被共享，优化了第一种继承方式的弊端，但是只能继承父类的实例属性和方法，不能继承原型属性或者方法

组合继承

function Person () {
    this.name = 'Person';
    this.play = [1, 2, 3];
}

Person.prototype.getName = function () {
    return this.name;
}
function Student() {
    // 第二次调用 Person()
    Person.call(this);
    this.type = 'Student';
}

// 第一次调用 Person()
Student.prototype = new Person();
// 手动挂上构造器，指向自己的构造函数
Student.prototype.constructor = Student;
var s3 = new Student();
var s4 = new Student();
s3.play.push(4);
console.log(s3.play, s4.play);  // 不互相影响
console.log(s3.getName()); // 正常输出'Person'
console.log(s4.getName()); // 正常输出'Person'
//引用属性不会被共享且能继承原型中的方法，但执行了两次，造成了多构造一次的性能开销

原型式继承

借助Object.create方法实现普通对象的继承，Object.create主要把传给这个方法的对象设为这个方法返回的对象的原型。

let parent4 = {
    name: "parent4",
    friends: ["p1", "p2", "p3"],
    getName: function() {
      return this.name;
    }
  };

  let person4 = Object.create(parent4);
  person4.name = "tom";
  person4.friends.push("jerry");

  let person5 = Object.create(parent4);
  person5.friends.push("lucy");

  console.log(person4.name); // tom
  console.log(person4.name === person4.getName()); // true
  console.log(person5.name); // parent4
  console.log(person4.friends); // ["p1", "p2", "p3","jerry","lucy"]
  console.log(person5.friends); // ["p1", "p2", "p3","jerry","lucy"]
//因为Object.create方法实现的是`浅拷贝`，多个实例的引用类型属性指向相同的内存，存在篡改的可能

寄生式继承

利用这个浅拷贝的能力再进行增强，添加一些方法

let parent5 = {
    name: "parent5",
    friends: ["p1", "p2", "p3"],
    getName: function() {
        return this.name;
    }
};
function clone(original) {
    let clone = Object.create(original);
    clone.getFriends = function() {
        return this.friends;
    };
    return clone;
}
let person5 = clone(parent5);
console.log(person5.getName()); // parent5
console.log(person5.getFriends()); // ["p1", "p2", "p3"]
//和原型式继承一样的问题

寄生组合式继承

属性都得到了继承，方法也没问题

function clone (parent, child) {
    // 这里改用 Object.create 就可以减少组合继承中多进行一次构造的过程
    child.prototype = Object.create(parent.prototype);
    child.prototype.constructor = child;
}

function Parent6() {
    this.name = 'parent6';
    this.play = [1, 2, 3];
}
Parent6.prototype.getName = function () {
    return this.name;
}
function Child6() {
    Parent6.call(this);
    this.friends = 'child5';
}

clone(Parent6, Child6);

Child6.prototype.getFriends = function () {
    return this.friends;
}

let person6 = new Child6();
console.log(person6); //{friends:"child5",name:"child5",play:[1,2,3],__proto__:Parent6}
console.log(person6.getName()); // parent6
console.log(person6.getFriends()); // child5

ES6中extends关键字

extends实际采用的也是寄生组合继承方式

class Car{
    constructor(color,speed){
        this.color = color
        this.speed = speed
    }
}
class Truck extends Car{
    constructor(color,speed){
        super(color,speed)
        this.color = "black" //覆盖
        this.Container = true // 货箱
    }
}
var truck=new Truck('red',10)
console.log(truck);//{ color: 'black', speed: 10, Container: true }

3.小结

通过Object.create 来划分不同的继承方式，最后的寄生式组合继承方式是通过组合继承改造之后的最优继承方式，而 extends 的语法糖和寄生组合继承的方式基本类似
