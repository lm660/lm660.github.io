手动实现数据结构-堆结构

1.堆结构

  • 堆的本质是一种特殊的树形数据结构，使用完全二叉树来实现:
  • 堆可以进行很多分类，但是平时使用的基本都是二叉堆;
  • 二叉堆又可以划分为最大堆和最小堆;
  • 最小堆:堆中每一个节点都小于等于它的子节点;
  • 最大堆:堆中每一个节点都大于等于它的子节点;

  • 堆结构通常是用来解决Top K问题的:
    Top k问题是指在一组数据中，找出最前面的K个最大/最小的元素;常用的解决方案有使用排序算法、快速选择算法、堆结构等;
  • 具体长什么样子，以及它是如何实现出来的:
    二叉堆用树形结构表示出来是一颗完全二叉树。通常在实现的时候我们底层会使用数组来实现

2.2插入操作-insert

//插入
  insert(value:T){
    //1.将元素放到数组尾部
    this.data.push(value)
    this.length++
    
    //2.维护最大堆特性（最后位置元素进行上滤操作）
    this.heapify_up()
  }
  //上滤操作
  heapify_up(){
    let index=this.length-1
    while(index>0){//直到比较到根节点（根节点索引为0）
      let parentIndex=Math.floor((index-1)/2)//父元素索引
    if(this.data[index]<=this.data[parentIndex]){//比较新元素与父元素的大小，若比父元素小就不用动了，
      break
    }
    //新元素大于父元素时，不需要else了，因为已经break跳出本次循环了。交换两个元素
    this.swip(index,parentIndex)
    index=parentIndex
    }
  }

2.3提取操作-extract

//提取
  extract():T|undefined{
    //1.判断长度为0和为1的情况
    if(this.length===0)return undefined
    if(this.length===1){
      this.length--
      return this.data.pop()!
    }
    //2.提取并且需要返回最大值
    const topValue=this.data[0]
    this.data[0]=this.data.pop()!//将最后一个元素放到第一个位置
    this.length--
    this.heapify_down()
    return topValue//第一个永远是最大值
    
  }
  //下滤操作函数
  private heapify_down(){
    //3.维护最大堆特征：下滤操作
    //定义索引位置
    let index=0
    while(2*index+1<this.length){//当左子索引存在就一直循环(因为是完全二叉树，左子索引不存在右子索引一定不存在)
     //找到左右子索引
    let leftChildIndex=2*index+1
    let rightChildIndex=leftChildIndex+1
    //找到左右子节点中较大值
    let largeIndex=leftChildIndex//先让左子索引为大
    if(rightChildIndex<this.length&&this.data[rightChildIndex]>this.data[leftChildIndex]){//若右子索引存在且右子元素大于左子元素
      largeIndex=rightChildIndex
    }
    //较大值和index值比较
    if(this.data[index]>=this.data[largeIndex]){//当前元素大于左右元素中最大的元素
      break
    }
    this.swip(index,largeIndex)//交换两个元素
    index=largeIndex//再将索引更新为找到的largeIndex
    }
  }

2.4 原地建堆-buildHeap

//原地建堆：不使用额外的内存空间直接在原有数组上操作
  buildHeap(arr:T[]){
    //1.使用arr
    this.data=arr
    this.length=arr.length
    //2.从第一个非叶子节点进行下滤操作
    const start=Math.floor((this.length-1)/2)
    for(let i=start;i>=0;i--){
      this.heapify_down(i)
    }
  }

2.5 二叉堆-最大/最小堆

传入true为最大堆，false为最小堆。前面的方法都是实现的最大堆

export default class Heap<T>{
   data:T[]=[]
  private length:number=0
  private isMax:boolean
  constructor(arr:T[]=[],isMax=true){
    this.isMax=isMax
    if(arr.length===0)return
    this.buildHeap(arr)
  }
    //
  private compare(i:number,j:number){
    if(this.isMax){
      return this.data[i]>=this.data[j]
    }else{
      return this.data[i]<=this.data[j]
    }
  }

  //常用到根据索引交换两个元素
  private swip(i:number,j:number){
   // [this.data[i],this.data[j]]=[this.data[j],this.data[i]]   //可以这样写的
   let temp=this.data[i]
   this.data[i]=this.data[j]
   this.data[j]=temp
  }
//插入
  insert(value:T){
    //1.将元素放到数组尾部
    this.data.push(value)
    this.length++
    
    //2.维护最大堆特性（最后位置元素进行上滤操作）
    this.heapify_up()
  }
  //上滤操作函数
  private heapify_up(){
    let index=this.length-1
    while(index>0){//直到比较到根节点（根节点索引为0）
      let parentIndex=Math.floor((index-1)/2)//父元素索引
      if(this.compare(parentIndex,index)){//比较新元素与父元素的大小，若比父元素大就不用动了，
    
      break
    }
    //新元素小于父元素时，不需要else了，因为已经break跳出本次循环了。交换两个元素
    this.swip(index,parentIndex)
    index=parentIndex
    }
  }
  //提取
  extract():T|undefined{
    //1.判断长度为0和为1的情况
    if(this.length===0)return undefined
    if(this.length===1){
      this.length--
      return this.data.pop()!
    }
    //2.提取并且需要返回最小值
    const topValue=this.data[0]
    this.data[0]=this.data.pop()!//将最后一个元素放到第一个位置
    this.length--
    this.heapify_down(0)
    return topValue//第一个永远是最小值
    
  }
  //下滤操作函数
  private heapify_down(start:number){
    //3.维护最大堆特征：下滤操作
    //定义索引位置
    let index=start
    while(2*index+1<this.length){//当左子索引存在就一直循环(因为是完全二叉树，左子索引不存在右子索引一定不存在)
     //找到左右子索引
    let leftChildIndex=2*index+1
    let rightChildIndex=leftChildIndex+1
    //找到左右子节点中较大值
    let largeIndex=leftChildIndex//先让左子索引为大
    if(rightChildIndex<this.length&&this.compare(rightChildIndex,leftChildIndex)){//若右子索引存在且右子元素大于左子元素 
      largeIndex=rightChildIndex
    }
    //较大值和index值比较
    if(this.compare(index,largeIndex)){//当前元素小于左右元素中最小的元素
    
      break
    }
    this.swip(index,largeIndex)//交换两个元素
    index=largeIndex//再将索引更新为找到的largeIndex
    }
  }
  peek():T|undefined{
    return this.data[0]
  }
  size():number{
    return this.length
  }
  isEmpty():boolean{
    return this.length===0
  }
  //原地建堆：不使用额外的内存空间直接在原有数组上操作
  buildHeap(arr:T[]){
    //1.使用arr
    this.data=arr
    this.length=arr.length
    //2.从第一个非叶子节点进行下滤操作
    const start=Math.floor((this.length-1)/2)
    for(let i=start;i>=0;i--){
      this.heapify_down(i)
    }
  }
}
