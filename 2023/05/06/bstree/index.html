手动实现数据结构-二叉搜索树

1.二叉搜索树

  • 二叉搜索树是一颗二叉树，可以为空;如果不为空，满足以下性质:

  • 非空左子树的所有键值小于其根节点的键值。

  • 非空右子树的所有键值大于其根节点的键值。

  • 左、右子树本身也都是二叉搜索树。

insert(value:T){
    const newNode=new treeNode(value)
    if(this.root===null){//第一次插入
      this.root=newNode
    }else{
      this.insertNode(this.root,newNode)
    }
  }
  private insertNode(node:treeNode<T>,newNode:treeNode<T>){
    if(newNode.value<node.value){//去左边继续查找空白位置
      if(node.left===null){//node节点的左边已经是空白，即可以插入了
        node.left=newNode
      }else{//否则，递归调用inserNode，不过是将node换成node.left
        this.insertNode(node.left,newNode)
      }
    }else{//去右边继续查找空白位置
      if(node.right===null){
        node.right=newNode
      }else{
        this.insertNode(node.right,newNode)
      }
    }
  }

search(value:T):boolean{
    let current=this.root//先将根节点设为当前节点
    while(current){//当current有值时就查找，直到返回true
      if(current.value===value){//假如当前节点等于查找的值返回true
        return true
      }
       if(current.value < value){//假如当前节点的值小于要查找的值，即要去右子树去查找
        current=current.right
      }else{//假如当前节点的值大于要查找的值，即要去左子树去查找
        current=current.left
      }
    }
    return false//查找所有节点都没有即没有找到返回false
  }

2.6搜索节点及搜索重构

private searchNode(value:T):treeNode<T>|null{
    let current=this.root//先将根节点设为当前节点
    let parent:treeNode<T>|null=null
    while(current){
      if(current.value===value){//如果找到返回
        return current
      }
      //继续向下找
      parent=current//先把父节点保存下来
       if(current.value < value){//假如当前节点的值小于要查找的值，即要去右子树去查找
        current=current.right
      }else{//假如当前节点的值大于要查找的值，即要去左子树去查找
        current=current.left
      }
      if(current)current.parent=parent//找到当前节点，当前节点的parent属性保存父节点
    }
   
    return null
  }

  //找后继节点:先获取其右子树，然后在右子树中找最小值
  private getsuccessor(delNode:treeNode<T>):treeNode<T>{
    let current=delNode.right//
    let successor:treeNode<T>|null=null//用来保存后继节点
    while(current){//只要current还有值，就是还没有找到最左边
      successor=current//先将当前节点设为后继节点
      current=current.left
      if(current){//为了保存父节点
        current.parent=successor
      }
    }
    console.log('删除节点：',delNode.value,'后继节点：',successor!.value);

    if(successor!==delNode.right){//相等的时候就不需要右连接了
      successor!.parent!.left=successor!.right//假如找到的后继节点还有一个右子节点，此时要将这个右子节点连接到successor父节点的左边
      successor!.right=delNode.right//进行右连接
    }
    //一定要进行的操作，将删除节点的left赋值给后继节点的left（就是用successor替换，要把要删除节点的左连接保留，连接到successor左边）
    successor!.left=delNode.left
    return successor!
  }

//删除必须拿到当前节点和父节点,以及current是左节点还是右节点
  remove(value:T):boolean{
    //1.搜索当前节点是否有value
    const current=this.searchNode(value)
    if(!current)return false
    console.log(current?.value,current?.parent?.value);
    //2.获取到当前节点、父节点、以及当前节点是父节点的左节点还是右节点
    //2.如果删除的是叶子节点：找到当前节点的父节点，然后根据当前节点是其父节点的左节点还是右节点，将其父节点的左节点/右节点设为null
    if(current.left===null&&current.right===null){//既没有左节点也没有右节点就是叶子节点
      if(current===this.root){//这个叶子节点也可能为根节点
        this.root=null
      }else if(current.isLeft){//当前节点是其父节点的左节点
        current.parent!.left=null
      }else{//当前节点是其父节点的右节点
        current.parent!.right=null
      }
    }

    //3.如果删除的节点有一个子节点
    //只有一个左节点：根据当前节点是其父节点的左节点还是右节点，将其父节点的左节点/右节点设为current.left(当前节点唯一左子节点)
    else if(current.right===null){
      if(current===this.root){//如果删除的是根节点，直接让唯一的左节点（即删除节点的子节点）设为根节点
        this.root=current.left
      }else if(current.isLeft){
        current.parent!.left=current.left
      }else{
        current.parent!.right=current.left
      }
    }
    //4.只有一个右节点，同理
    else if(current.left===null){
      if(current===this.root){
        this.root=current.right
      }else if(current.isLeft){
        current.parent!.left=current.right
      }else{
        current.parent!.right=current.right
      }
    }
    //5.如果删除的节点有2个子节点
    //要从下面的子节点中找到一个节点替换当前节点，这个节点应该是最接近current的（要么左子树中最大值-前驱，要么右子树中最小值-后继）
    //在此找后继
    else{//总共就5种情况
      const successor= this.getsuccessor(current)
      if(current===this.root){
        this.root=successor
      }else if(current.isLeft){
        current.parent!.left=successor
      }else{
        current.parent!.right=successor
      }
    }
    return true
  }

   function BinarySerachTree() {
       // 创建节点构造函数
       function Node(key) {
           this.key = key
           this.left = null
           this.right = null
       }

       // 保存根的属性
       this.root = null

       // 二叉搜索树相关的操作方法
       // 向树中插入数据
       BinarySerachTree.prototype.insert = function (key) {
           // 1.根据key创建对应的node
           var newNode = new Node(key)

           // 2.判断根节点是否有值
           if (this.root === null) {
               this.root = newNode
           } else {
               this.insertNode(this.root, newNode)
           }
       }

       BinarySerachTree.prototype.insertNode = function (node, newNode) {
           if (newNode.key < node.key) { // 1.准备向左子树插入数据
               if (node.left === null) { // 1.1.node的左子树上没有内容
                   node.left = newNode
               } else { // 1.2.node的左子树上已经有了内容
                   this.insertNode(node.left, newNode)
               }
           } else { // 2.准备向右子树插入数据
               if (node.right === null) { // 2.1.node的右子树上没有内容
                   node.right = newNode
               } else { // 2.2.node的右子树上有内容
                   this.insertNode(node.right, newNode)
               }
           }
       }

       // 获取最大值和最小值
       BinarySerachTree.prototype.min = function () {
           var node = this.root
           while (node.left !== null) {
               node = node.left
           }
           return node.key
       }

       BinarySerachTree.prototype.max = function () {
           var node = this.root

       // 删除结点
       BinarySerachTree.prototype.remove = function (key) {
           // 1.定义临时保存的变量
           var current = this.root
           var parent = this.root
           var isLeftChild = true

           // 2.开始查找节点
           while (current.key !== key) {
               parent = current
               if (key < current.key) {
                   isLeftChild = true
                   current = current.left
               } else {
                   isLeftChild = false
                   current = current.right
               }

               // 如果发现current已经指向null, 那么说明没有找到要删除的数据
               if (current === null) return false
           }

           // 3.删除的结点是叶结点
           if (current.left === null && current.right === null) {
               if (current == this.root) {
                   this.root == null
               } else if (isLeftChild) {
                   parent.left = null
               } else {
                   parent.right = null
               }
           }

           // 4.删除有一个子节点的节点
           else if (current.right === null) {
               if (current == this.root) {
                   this.root = current.left
               } else if (isLeftChild) {
                   parent.left = current.left
               } else {
                   parent.right = current.left
               }
           } else if (current.left === null) {
               if (current == this.root) {
                   this.root = current.right
               } else if (isLeftChild) {
                   parent.left = current.right
               } else {
                   parent.right = current.right
               }
           }

           // 5.删除有两个节点的节点
           else {
               // 1.获取后继节点
               var successor = this.getSuccessor(current)

               // 2.判断是否是根节点
               if (current == this.root) {
                   this.root = successor
               } else if (isLeftChild) {
                   parent.left = successor
               } else {
                   parent.right = successor
               }

               // 3.将删除节点的左子树赋值给successor
               successor.left = current.left
           }

           return true
       }

       // 找后继的方法
       BinarySerachTree.prototype.getSuccessor = function (delNode) {
           // 1.使用变量保存临时的节点
           var successorParent = delNode
           var successor = delNode
           var current = delNode.right // 要从右子树开始找

           // 2.寻找节点
           while (current != null) {
               successorParent = successor
               successor = current
               current = current.left

           // 3.如果是删除图中15的情况, 还需要如下代码

               this.postOrderTraversalNode(node.right, handler)
