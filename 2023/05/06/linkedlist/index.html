手动实现数据结构-链表结构

1.链表结构

  • 都知道数组是一个顺序储存方式的数据结构，但是，数组最大的缺点就是我们的插入和删除时需要移动大量的元素，浪费性能。并且必须为数组开足够的空间，否则有溢出风险。

  • 链表通过不连续的储存方式，自适应内存大小，以及指针的灵活使用，巧妙的简化了上述的问题。

  • 在插入和删除操作时，时间复杂度O(1)，相对数组效率高很多，但是访问任何一个位置元素都需要从头开始访问，无法通过下标直接访问元素。

//定义两个类，一个是链表中的节点类便于链表操作，一个是链表类用于创建链表实例
//根据链表定义，节点中包含两个元素，一个是存放的数据，一个是next用于指向下一个节点
class Node<T>{
value:T
next:Node<T>|null=null//next的值可能是节点也可能为null，赋初始值为null
constructor(value:T){

}//因为不清楚value初值是什么样的，通过这种方法赋值

class LinkedList<T>{

  size:number=0//链表长度初始为0
  get length(){
    return this.size
  }//类比数组，前面加get，调用的时候就可.length,不用.length()

2.2添加-append

class Node<T>{
value:T
next:Node<T>|null=null
constructor(value:T){
  this.value=value
}
}

class LinkedList<T>{
 private head:Node<T>|null=null
 private size:number=0
  get length(){
    return this.size
  }

  //添加（追加)方法
  append(value:T){
    //首先需要创建一个节点
    const newnode=new Node<T>(value)
    //添加节点的时候需要判断链表本身为不为空，
    if(!this.head){
      this.head=newnode//为空的时候直接让this.head指向新创建的节点
    }else{//不为空的时候，定义一个临时变量表示当前节点，一直判断当前节点的next值为不为空，不为空就让当前节点的next赋为当前节点，为空代表找到了链表的最后一个节点，
      let currentnode=this.head
      while(currentnode.next){
        currentnode=currentnode.next
      }
      currentnode.next=newnode//找到了最后一个节点就让他的next指向新创建的节点
    }
    this.size++
  }

  //遍历方法
  traverse(){
    const arr:T[]=[]
    let current=this.head//定义一个当前节点变量
    while(current){//只要当前变量有值就一直循环，直到遍历完整个链表
      arr.push(current.value)//把当前节点的值放到数组中
      current=current.next//再把当前节点的next设为当前节点
    }
    return arr.join('->')
  }
}

linkedList.append('aaa')
linkedList.append('nnn')
console.log(linkedList.traverse());//aaa->nnn

2.3插入-insert

insert(value:T,position:number):boolean{
    //边界判断
    if(position<0||position>this.size)return false
    const newnode=new Node(value)
    //判断插入的位置是否为0
    if(position===0){
      newnode.next=this.head//必须先让新节点指向第一个节点（即为this.head）
      this.head=newnode//再让this.head指向新节点
    }else{//主要是找到插入位置的节点及前一个节点
      let previous:Node<T>|null=null
      let index=0//控制指向次数直到达到position
      let current=this.head
      while(index++<position){
        previous=current
        current=current!.next//current有可能为null，在这个位置一定有值，因为前面已经进行边界判断了，所以可以用断言
      }
      //index=position
      newnode.next=current//新节点指向current
      previous!.next=newnode//先前节点指向新节点，
    }
    this.size++
    return true
  }

2.4根据位置删除-removeAt

//删除方法
removeAt(position:number):T|null{
  //越界判断
  if(position<0||position>=this.size)return null

  //判断是不是删除第一个
  let current=this.head
  if(position==0){
    this.head=current?.next??null//表示current可能为undefined时，让值为null
  }else{
    let previous:Node<T>|null=null
    let index=0
    while(index++<position&&current){
      previous=current
      current=current.next
    }
    previous!.next=current?.next??null
  }
  this.size--
  return current?.value??null

}

2.6根据位置修改元素-update

//根据位置获取节点（对代码进行重构，因为很多用到根据位置获取节点，然后再进行操作）
private getNode(position:number):Node<T>|null{
  if(position<0||position>=this.size)return null
  let current=this.head
  let index=0
  while(index++<position&&current){
    current=current.next
  }
  return current
}

  //get方法根据位置获取元素（重构之后）
  get(position:number):T|null{
    if(position<0||position>=this.length)return null

    return this.getNode(position)?.value??null
  }
     //删除方法
  removeAt(position:number):T|null{
    //越界判断
    if(position<0||position>=this.length)return null

    //判断是不是删除第一个
    let current=this.head
    if(position==0){
      this.head=current?.next??null//表示current可能为undefined时，让值为null
    }else{
      const previous=this.getNode(position-1)
      //需要给current重新赋值，否则返回的值永远是第一个值
      current=previous!.next
      previous!.next=previous?.next?.next??null
    }
    this.length--
    return current?.value??null

  }

  //根据位置修改元素
  update(value:T,position:number):boolean{
    if(position<0||position>=this.length)return false
    //获取指定位置节点
    let currentnode=this.getNode(position)
    currentnode!.value=value
    return true
  }

3.使用JS实现

// 封装链表的构造函数
   function LinkedList() {
       // 封装一个Node类, 用于保存每个节点信息
       function Node(element) {
           this.element = element
           this.next = null
       }

       // 链表中的属性
       this.length = 0
       this.head = null

       // 链表尾部追加元素方法
       LinkedList.prototype.append = function (element) {
           // 1.根据新元素创建节点
           var newNode = new Node(element)

           // 2.判断原来链表是否为空
           if (this.head === null) { // 链表尾空
               this.head = newNode
           } else { // 链表不为空
               // 2.1.定义变量, 保存当前找到的节点
               var current = this.head
               while (current.next) {
                   current = current.next
               }

               // 2.2.找到最后一项, 将其next赋值为node
               current.next = newNode
           }

           // 3.链表长度增加1
           this.length++
       }

       // 链表的toString方法
       LinkedList.prototype.toString = function () {
           // 1.定义两个变量
           var current = this.head
           var listString = ""

           // 2.循环获取链表中所有的元素
           while (current) {
               listString += "," + current.element
               current = current.next
           }

           // 3.返回最终结果
           return listString.slice(1)
       }

       // 根据下标删除元素
       LinkedList.prototype.insert = function (position, element) {
           // 1.检测越界问题: 越界插入失败
           if (position < 0 || position > this.length) return false

           // 2.定义变量, 保存信息
           var newNode = new Node(element)
           var current = this.head
           var previous = null
           index = 0

           // 3.判断是否列表是否在第一个位置插入
           if (position == 0) {
               newNode.next = current
               this.head = newNode
           } else {
               while (index++ < position) {
                   previous = current
                   current = current.next
               }

               newNode.next = current
               previous.next = newNode
           }

           // 4.length+1
           this.length++

           return true
       }

       // 根据位置移除节点
       LinkedList.prototype.removeAt = function (position) {
           // 1.检测越界问题: 越界移除失败, 返回null
           if (position < 0 || position >= this.length) return null

           // 2.定义变量, 保存信息
           var current = this.head
           var previous = null
           var index = 0

           // 3.判断是否是移除第一项
           if (position === 0) {
               this.head = current.next
           } else {
               while (index++ < position) {
                   previous = current
                   current = current.next
               }

               previous.next = current.next
           }

           // 4.length-1
           this.length--

           // 5.返回移除的数据
           return current.element
       }

       // 根据元素获取链表中的位置
       LinkedList.prototype.indexOf = function (element) {
           // 1.定义变量, 保存信息
           var current = this.head
           index = 0

           // 2.找到元素所在的位置
           while (current) {
               if (current.element === element) {
                   return index
               }
               index++
               current = current.next
           }

           // 3.来到这个位置, 说明没有找到, 则返回-1
           return -1
       }

       //根据位置获取元素
       LinkedList.prototype.get=function(position){
           //首先越级判断
           if(position<0||position>=this.length) return null
           //获取对应元素
           var index=0
           var current=this.head
           while(index++<position){
               current=current.next
           }
           return current.data
       }

       // 根据元素删除信息
       LinkedList.prototype.remove = function (element) {
           var index = this.indexOf(element)
           return this.removeAt(index)
       }

       // 判断链表是否为空
       LinkedList.prototype.isEmpty = function () {
           return this.length == 0
       }

       // 获取链表的长度
       LinkedList.prototype.size = function () {
           return this.length
       }

       // 获取第一个节点
       LinkedList.prototype.getFirst = function () {
           return this.head.element
       }
   }

4.2删除链表中的节点（力扣237）

 class ListNode {
       val: number
       next: ListNode | null
       constructor(val?: number, next?: ListNode | null) {
           this.val = (val===undefined ? 0 : val)
           this.next = (next===undefined ? null : next)
       }
   }
//不想着把当前节点删除，因为也访问不到第一个节点。换个思路，把下一个节点的值给当前节点，把当前节点指向下下节点，相当于把当前节点的下一个节点删除了
function deleteNode(node: ListNode | null): void {
  node!.val=node!.next!.val
  node!.next=node!.next!.next
 };

4.3反转链表（力扣206）

class ListNode {
  val: number
  next: ListNode | null
  constructor(val?: number, next?: ListNode | null) {
      this.val = (val===undefined ? 0 : val)
      this.next = (next===undefined ? null : next)
  }
}


//迭代
function reverseList(head: ListNode | null): ListNode | null {
  let pre:ListNode|null=null//存储前一个节点
  let cur=head//存储当前节点
  while(cur){//直至到链表末尾
      const next=cur.next//先保存当前节点的下一节点，不先保存，后面的节点就拿不到了，下面会让当前节点指向上一节点，后面再使用cur.next指的就是上一节点了
      cur.next=pre//让当前节点指向前一个节点
      pre=cur//把前一个节点移一位
      cur=next//同时把当前节点也移一位
  }
  return pre//返回新的链表头
};

//递归方法
function reverseList1(head: ListNode | null): ListNode | null {
  if (head == null || head.next == null) {
      return head;
  }//递归停止条件
  const newHead = reverseList1(head.next);//为了到达最后一个节点
  //完成想要做的操作在这个位置 第一次head为倒数第二个节点，如果的倒数第一个节点就直接返回head了
  head.next.next = head;//让最后一个节点指向前一个
  head.next = null;//前一个指向null
  return newHead;
};
