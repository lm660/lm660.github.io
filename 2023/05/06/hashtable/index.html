手动实现数据结构-哈希表

1.哈希表

  • 基于数组实现，可提供快速的插入-删除-查找，复杂度O(1),比树结构还快。

  • 但是数据是无序的，key是不重复的。哈希表是基于数组的，我们知道数组创建后扩容成本比较高，所以当哈希表被填满时，性能下降的比较严重。

  • 就是数组结构，但在于对数组下标值的一种变换（哈希函数）

1.1相关概念

  • 哈希化：将大数字转化为数组范围内下标的过程（1.幂的连乘2.求余操作）

  • 哈希函数：将单词转化为大数字，大数字再进行哈希化的函数。哈希函数要具有2个特点：

快速：使用霍纳法则降低时间复杂度。均匀分布：将哈希表长度、n次幂底数均使用质数。（因为质数容易产生唯一性结果，减少冲突）

  • 冲突：不同的key通过同一哈希函数产生了相同的哈希位置。

解决冲突：

链地址法：每个数组元素中存储的不再是单个数据，而是一个链条

2.2插入/修改-put

class HashTable<T>{
   storage:[string,T][][]=[]
  private length:number=7
  private count:number=0
  //根据key获取index值(也就是哈希函数)
  private getIndex(key:string,max:number):number{
    const length=key.length
    let hashcode:number=0
    for(let i=0;i<length;i++){
      hashcode=31*hashcode+key.charCodeAt(i)
    }
    let index=hashcode%max
    return index
  }

  //插入/修改
  put(key:string,value:T){
    //首先根据key值获取数组中对应的index
    let index=this.getIndex(key,this.length)
    //在获取这个位置对应的数组（桶）
    let bucket=this.storage[index]

    //判断bucket是否有值，首次插入都是undefined，
    if(!bucket){
      bucket=[]
      this.storage[index]=bucket
    }
    //记录是否修改
    let isUpdata=false
    //bucket有值情况下遍历bucket,确定里面有没有key

    for(let i=0;i<bucket.length;i++){
      //获取里面的每一项元组
      const tuple=bucket[i]
      const tuplekey=tuple[0]
      let tuplevalue=tuple[1]
      if(tuplekey===key){
        tuplevalue=value
        isUpdata=true
      }

    }
    //遍历完了也没找到key，说明该位置上面没有，进行添加
    if(!isUpdata){
      bucket.push([key,value])//将添加的键值对数据添加到bucket
      this.count++
    }

  }
}

    const index=this.getIndex(key,this.length)

    let bucket=this.storage[index]

    //遍历bucket所有项查找key
    for(let i=0;i<bucket.length;i++){
      const tuple=bucket[i]
      const tuplekey=tuple[0]
      const tuplevalue=tuple[1]
      if(tuplekey===key){
        return tuplevalue
      }
    }
    return undefined
  }

 delete(key:string):T|undefined{
   const index=this.getIndex(key,this.length)
   const bucket=this.storage[index]
   if(!bucket)return undefined
   for(let i=0;i<bucket.length;i++){
     const tuple=bucket[i]

     const tuplevalue=tuple[1]
     if(tuplekey===key){
        bucket.splice(i,1)
        this.count--

     }

private isPrime(num:number):boolean{
    //O(n)=>O(logn)遍历到平方根
    const sqrt=Math.sqrt(num)
    for(let i=2;i<=sqrt;i++){
      if(num%i==0){
        return false
      }
    }
    return true
  }

3.使用JS实现

// 创建HashTable构造函数
  function HashTable() {
      // 定义属性
      this.storage = []
      this.count = 0
      this.limit = 8

      // 定义相关方法
      // 判断是否是质数
      HashTable.prototype.isPrime = function (num) {
          var temp = parseInt(Math.sqrt(num))
          // 2.循环判断
          for (var i = 2; i <= temp; i++) {
              if (num % i == 0) {
                  return false
              }
          }
          return true
      }

      // 获取质数
      HashTable.prototype.getPrime = function (num) {
          while (!isPrime(num)) {
              num++
          }
          return num
      }

      // 哈希函数
      HashTable.prototype.hashFunc = function(str, max) {
          // 1.初始化hashCode的值
          var hashCode = 0

          // 2.霍纳算法, 来计算hashCode的数值
          for (var i = 0; i < str.length; i++) {
              hashCode = 37 * hashCode + str.charCodeAt(i)
          }

          // 3.取模运算
          hashCode = hashCode % max
          return hashCode
      }

      // 插入数据方法
      HashTable.prototype.put = function (key, value) {
          // 1.获取key对应的index
          var index = this.hashFunc(key, this.limit)

          // 2.取出数组(也可以使用链表)
          // 数组中放置数据的方式: [[ [k,v], [k,v], [k,v] ] , [ [k,v], [k,v] ]  [ [k,v] ] ]
          var bucket = this.storage[index]

          // 3.判断这个数组是否存在
          if (bucket === undefined) {
              // 3.1创建桶
              bucket = []
              this.storage[index] = bucket
          }

          // 4.判断是新增还是修改原来的值.
          var override = false
          for (var i = 0; i < bucket.length; i++) {
              var tuple = bucket[i]
              if (tuple[0] === key) {
                  tuple[1] = value
                  override = true
              }
          }

          // 5.如果是新增, 前一步没有覆盖
          if (!override) {
              bucket.push([key, value])
              this.count++

              if (this.count > this.limit * 0.75) {
                  var primeNum = this.getPrime(this.limit * 2)
                  this.resize(primeNum)
              }
          }
      }

      // 获取存放的数据
      HashTable.prototype.get = function (key) {
          // 1.获取key对应的index
          var index = this.hashFunc(key, this.limit)

          // 2.获取对应的bucket
          var bucket = this.storage[index]

          // 3.如果bucket为null, 那么说明这个位置没有数据
          if (bucket == null) {
              return null
          }

          // 4.有bucket, 判断是否有对应的key
          for (var i = 0; i < bucket.length; i++) {
              var tuple = bucket[i]
              if (tuple[0] === key) {
                  return tuple[1]
              }
          }

          // 5.没有找到, return null
          return null
      }

      // 删除数据
      HashTable.prototype.remove = function (key) {
          // 1.获取key对应的index
          var index = this.hashFunc(key, this.limit)

          // 2.获取对应的bucket
          var bucket = this.storage[index]

          // 3.判断同是否为null, 为null则说明没有对应的数据
          if (bucket == null) {
              return null
          }

          // 4.遍历bucket, 寻找对应的数据
          for (var i = 0; i < bucket.length; i++) {
              var tuple = bucket[i]
              if (tuple[0] === key) {
                  bucket.splice(i, 1)
                  this.count--

                  // 缩小数组的容量
                  if (this.limit > 7 && this.count < this.limit * 0.25) {
                      var primeNum = this.getPrime(Math.floor(this.limit / 2))
                      this.resize(primeNum)
                  }
              }
              return tuple[1]
          }

          // 5.来到该位置, 说明没有对应的数据, 那么返回null
          return null
      }

      // isEmpty方法
      HashTable.prototype.isEmpty = function () {
          return this.count == 0
      }

      // size方法
      HashTable.prototype.size = function () {
          return this.count
      }

      // 哈希表扩容
      HashTable.prototype.resize = function (newLimit) {
          // 1.保存旧的数组内容
          var oldStorage = this.storage

          // 2.重置属性
          this.limit = newLimit
          this.count = 0
          this.storage = []

          // 3.遍历旧数组中的所有数据项, 并且重新插入到哈希表中
          oldStorage.forEach(function (bucket) {
              // 1.bucket为null, 说明这里面没有数据
              if (bucket == null) {
                  return
              }

              // 2.bucket中有数据, 那么将里面的数据重新哈希化插入
              for (var i = 0; i < bucket.length; i++) {
                  var tuple = bucket[i]
                  this.put(tuple[0], tuple[1])
              }
          }).bind(this)
      }
  }
